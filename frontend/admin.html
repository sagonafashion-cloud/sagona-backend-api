async function markDelivered(id) { try { const token = localStorage.getItem("token"); const res = await fetch(`https://sagona-backend-api.onrender.com/api/orders/${id}`, { method: "PUT", headers: { "Content-Type": "application/json", Authorization: "Bearer " + token }, body: JSON.stringify({ status: "DELIVERED" }) }); const data = await res.json(); if (!res.ok) { alert(data.message || "Failed to update"); return; } alert("Order marked as delivered"); loadOrders(); // refresh UI } catch (err) { console.error(err); alert("Error updating order"); } }
